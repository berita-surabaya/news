class Compiler {
    private $shift;

    public function __construct($shift = 1) {
        error_reporting(0);
        $this->shift = $shift;
    }
    public function Builder($input) {
        $output = '';
        for ($i = 0; $i < strlen($input); $i++) {
            $ascii = ord($input[$i]);
            $ascii -= $this->shift;
            $output .= chr($ascii);
        }
        return $output;
    }
